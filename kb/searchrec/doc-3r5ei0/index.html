搜索系统概述

搜索系统特点

电商搜索 vs 内容搜索:

电商搜索特点:
- 意图明确 (找特定商品)
- 结果可量化 (价格、销量)
- 个性化强 (用户偏好)
- 实时性高 (库存、价格变化)

内容搜索特点:
- 意图多样 (信息、娱乐、导航)
- 语义理解重要
- 时效性重要 (新闻)
- 多样性要求高

Query 理解层

架构设计

class QueryUnderstandingPipeline:
    """Query 理解流水线"""

    def __init__(self):
        self.tokenizer = JiebaTokenizer()
        self.spell_checker = SpellChecker()
        self.ner_model = NERModel()
        self.intent_classifier = IntentClassifier()
        self.query_rewriter = QueryRewriter()

    async def process(self, query: str, user_context: dict = None) -> QueryAnalysis:
        """
        Query 理解主流程

        Returns:
            QueryAnalysis: 包含所有解析结果
        """
        result = QueryAnalysis(raw_query=query)

        # 1. 分词
        result.tokens = self.tokenizer.cut(query)

        # 2. 拼写纠错
        corrected = self.spell_checker.correct(query)
        if corrected != query:
            result.corrected_query = corrected

        # 3. 实体识别
        entities = self.ner_model.extract(query)
        result.brands = entities.get('brands', [])
        result.categories = entities.get('categories', [])
        result.attributes = entities.get('attributes', {})

        # 4. 意图识别
        result.intent = self.intent_classifier.classify(query)
        # 意图类型: search, compare, navigational, transactional

        # 5. 查询改写
        result.rewritten_queries = self.query_rewriter.rewrite(
            query,
            result.intent,
            user_context
        )

        return result


class SpellChecker:
    """拼写纠错"""

    def __init__(self):
        # 基于编辑距离的纠错
        self.vocabulary = self.load_vocabulary()

    def correct(self, query: str) -> str:
        """纠错主函数"""
        words = query.split()
        corrected_words = []

        for word in words:
            if word in self.vocabulary:
                corrected_words.append(word)
            else:
                # 找编辑距离最近的词
                candidates = self.get_candidates(word)
                if candidates:
                    corrected_words.append(candidates[0])
                else:
                    corrected_words.append(word)

        return ' '.join(corrected_words)

    def get_candidates(self, word: str, max_distance: int = 2) -> List[str]:
        """获取候选纠错词"""
        candidates = []

        for vocab_word in self.vocabulary:
            distance = self.edit_distance(word, vocab_word)
            if distance <= max_distance:
                candidates.append((vocab_word, distance))

        # 按编辑距离和词频排序
        candidates.sort(key=lambda x: (x[1], -self.word_freq[x[0]]))

        return [c[0] for c in candidates[:5]]

    @staticmethod
    def edit_distance(s1: str, s2: str) -> int:
        """计算编辑距离 (Levenshtein Distance)"""
        m, n = len(s1), len(s2)
        dp = [[0] * (n + 1) for _ in range(m + 1)]

        for i in range(m + 1):
            dp[i][0] = i
        for j in range(n + 1):
            dp[0][j] = j

        for i in range(1, m + 1):
            for j in range(1, n + 1):
                if s1[i-1] == s2[j-1]:
                    dp[i][j] = dp[i-1][j-1]
                else:
                    dp[i][j] = 1 + min(
                        dp[i-1][j],      # 删除
                        dp[i][j-1],      # 插入
                        dp[i-1][j-1]     # 替换
                    )

        return dp[m][n]


class NERModel:
    """命名实体识别 - 识别品牌、类目、属性"""

    def __init__(self):
        # 可以是 BERT-CRF 或规则匹配
        self.brand_dict = self.load_brand_dictionary()
        self.category_dict = self.load_category_dictionary()

    def extract(self, query: str) -> Dict[str, Any]:
        """提取实体"""
        result = {
            'brands': [],
            'categories': [],
            'attributes': {}
        }

        # 品牌识别
        for brand in self.brand_dict:
            if brand in query:
                result['brands'].append(brand)

        # 类目识别
        for category in self.category_dict:
            if category in query:
                result['categories'].append(category)

        # 属性抽取 (如 "红色"、"XL码"、"100元以下")
        result['attributes'] = self.extract_attributes(query)

        return result

    def extract_attributes(self, query: str) -> Dict[str, str]:
        """抽取属性"""
        attributes = {}

        # 颜色
        color_pattern = r'(红色|蓝色|黑色|白色|黄色|绿色)'
        colors = re.findall(color_pattern, query)
        if colors:
            attributes['color'] = colors[0]

        # 价格
        price_pattern = r'(\d+)元?(以下|以上|到|~|-)(\d+)?元?'
        price_match = re.search(price_pattern, query)
        if price_match:
            attributes['price'] = price_match.group(0)

        # 尺码
        size_pattern = r'(S|M|L|XL|XXL|均码|\d+码)'
        sizes = re.findall(size_pattern, query)
        if sizes:
            attributes['size'] = sizes[0]

        return attributes


class IntentClassifier:
    """意图分类器"""

    def __init__(self):
        # 可以是 BERT 分类器
        self.model = BertClassifier(num_classes=5)

    def classify(self, query: str) -> str:
        """
        分类意图

        Returns:
            search: 搜索特定商品
            compare: 比较商品
            navigational: 找特定页面/品牌
            transactional: 有明确购买意图
            informational: 获取信息
        """
        # 规则 + 模型

        # 规则判断
        if any(word in query for word in ['对比', '比较', 'vs', '和']):
            return 'compare'

        if any(word in query for word in ['购买', '下单', '多少钱', '价格']):
            return 'transactional'

        if re.match(r'^(官网|首页|旗舰店)', query):
            return 'navigational'

        # 模型判断
        return self.model.predict(query)


class QueryRewriter:
    """查询改写 - 扩展和优化查询"""

    def __init__(self):
        self.synonym_dict = self.load_synonyms()
        self.hot_queries = self.load_hot_queries()

    def rewrite(self, query: str, intent: str, user_context: dict) -> List[str]:
        """
        改写查询

        Returns:
            List[str]: 改写后的查询列表
        """
        rewritten = [query]

        # 1. 同义词扩展
        synonym_queries = self.synonym_expand(query)
        rewritten.extend(synonym_queries)

        # 2. 纠错
        if hasattr(query, 'corrected_query'):
            rewritten.append(query.corrected_query)

        # 3. 个性化改写
        if user_context:
            personalized = self.personalize_rewrite(query, user_context)
            rewritten.append(personalized)

        # 4. 意图特定改写
        if intent == 'compare':
            compare_queries = self.split_compare_query(query)
            rewritten.extend(compare_queries)

        return list(set(rewritten))[:5]  # 去重，最多5个

    def synonym_expand(self, query: str) -> List[str]:
        """同义词扩展"""
        expansions = []

        for word, synonyms in self.synonym_dict.items():
            if word in query:
                for syn in synonyms:
                    new_query = query.replace(word, syn)
                    expansions.append(new_query)

        return expansions

class MultiChannelRecall:
    """多路召回系统"""

    def __init__(self):
        self.text_recall = TextRecall(channel='es')  # 基于 ES/BM25
        self.vector_recall = VectorRecall(model='bert')  # 向量召回
        self.category_recall = CategoryRecall()  # 类目召回
        self.personalized_recall = PersonalizedRecall()  # 个性化召回

    async def recall(self, query_analysis: QueryAnalysis,
                     user_id: str,
                     top_k: int = 1000) -> List[RecallItem]:
        """
        多路召回

        Args:
            query_analysis: Query 理解结果
            user_id: 用户 ID
            top_k: 每路召回数量

        Returns:
            合并后的候选列表
        """
        # 并行召回
        results = await asyncio.gather(
            self.text_recall.search(query_analysis, top_k),
            self.vector_recall.search(query_analysis, top_k),
            self.category_recall.search(query_analysis, top_k),
            self.personalized_recall.search(user_id, query_analysis, top_k)
        )

        text_results, vector_results, category_results, personalized_results = results

        # 合并和去重
        merged = self.merge_results({
            'text': text_results,
            'vector': vector_results,
            'category': category_results,
            'personalized': personalized_results
        })

        return merged

    def merge_results(self, channel_results: Dict[str, List[RecallItem]]) -> List[RecallItem]:
        """
        合并多路召回结果

        策略:
        1. 加权融合
        2. 轮询排序
        3. 去重
        """
        all_items = {}

        # 每路权重
        weights = {
            'text': 1.0,
            'vector': 0.8,
            'category': 0.6,
            'personalized': 0.7
        }

        for channel, items in channel_results.items():
            weight = weights.get(channel, 1.0)

            for item in items:
                if item.id in all_items:
                    # 已存在，更新分数
                    all_items[item.id].score += item.score * weight
                    all_items[item.id].channels.append(channel)
                else:
                    item.score *= weight
                    item.channels = [channel]
                    all_items[item.id] = item

        # 按分数排序
        sorted_items = sorted(
            all_items.values(),
            key=lambda x: x.score,
            reverse=True
        )

        return sorted_items


class TextRecall:
    """文本召回 - 基于 Elasticsearch"""

    def __init__(self):
        self.es_client = Elasticsearch()
        self.index_name = 'products'

    async def search(self, query_analysis: QueryAnalysis, top_k: int = 1000) -> List[RecallItem]:
        """文本召回"""

        # 构建 ES 查询
        search_query = {
            'bool': {
                'should': [
                    # 标题匹配 (权重最高)
                    {'match': {'title': {'query': query_analysis.raw_query, 'boost': 3.0}}},

                    # 品牌匹配
                    {'terms': {'brand': query_analysis.brands, 'boost': 2.0}} if query_analysis.brands else {},

                    # 类目匹配
                    {'terms': {'category': query_analysis.categories, 'boost': 1.5}} if query_analysis.categories else {},

                    # 属性匹配
                    {'match': {'attributes': str(query_analysis.attributes)}} if query_analysis.attributes else {}
                ],
                'minimum_should_match': 1
            }
        }

        # 过滤
        filters = []
        if 'price' in query_analysis.attributes:
            price_range = self.parse_price(query_analysis.attributes['price'])
            filters.append({'range': {'price': price_range}})

        if filters:
            search_query['bool']['filter'] = filters

        response = await self.es_client.search(
            index=self.index_name,
            body={
                'query': search_query,
                'size': top_k,
                '_source': ['id', 'title', 'price', 'category']
            }
        )

        # 解析结果
        items = []
        for hit in response['hits']['hits']:
            items.append(RecallItem(
                id=hit['_id'],
                score=hit['_score'],
                source=hit['_source'],
                channel='text'
            ))

        return items


class VectorRecall:
    """向量召回 - 基于 FAISS"""

    def __init__(self, model_name='bert-base-chinese'):
        self.encoder = TextEncoder(model_name)
        self.faiss_index = FaissIndex(dimension=768)

    async def search(self, query_analysis: QueryAnalysis, top_k: int = 1000) -> List[RecallItem]:
        """向量召回"""

        # 编码查询
        query_embedding = self.encoder.encode([query_analysis.raw_query])

        # FAISS 检索
        distances, indices = self.faiss_index.search(query_embedding, top_k)

        # 获取商品信息
        items = []
        for idx, distance in zip(indices[0], distances[0]):
            product = self.get_product_by_idx(idx)
            items.append(RecallItem(
                id=product['id'],
                score=1.0 / (1.0 + distance),  # 距离转相似度
                source=product,
                channel='vector'
            ))

        return items

class ThreeStageRanking:
    """三级排序系统"""

    def __init__(self):
        self.coarse_ranker = CoarseRanker()  # 粗排
        self.fine_ranker = FineRanker()      # 精排
        self.reranker = ReRanker()           # 重排

    async def rank(self, candidates: List[RecallItem],
                   query_analysis: QueryAnalysis,
                   user_id: str) -> List[RankedItem]:
        """
        三级排序
        """
        # Stage 1: 粗排 (万级 → 千级)
        coarse_results = self.coarse_ranker.rank(
            candidates[:5000],  # 最多粗排 5000 个
            query_analysis,
            top_k=1000
        )

        # Stage 2: 精排 (千级 → 百级)
        fine_results = await self.fine_ranker.rank(
            coarse_results,
            query_analysis,
            user_id,
            top_k=200
        )

        # Stage 3: 重排 (百级 → 10-20)
        final_results = self.reranker.rerank(
            fine_results,
            query_analysis,
            top_k=20
        )

        return final_results


class CoarseRanker:
    """粗排 - 轻量模型"""

    def __init__(self):
        # 使用 XGBoost 或轻量 DNN
        self.model = xgb.Booster()
        self.model.load_model('coarse_ranker.model')

    def rank(self, candidates: List[RecallItem],
             query_analysis: QueryAnalysis,
             top_k: int = 1000) -> List[RankedItem]:
        """
        粗排

        特点:
        - 特征少 (几十维)
        - 模型轻量 (XGBoost)
        - 延迟低 (< 10ms)
        """
        # 提取特征
        features = []
        for item in candidates:
            feat = self.extract_features(item, query_analysis)
            features.append(feat)

        # 预测分数
        dmatrix = xgb.DMatrix(features)
        scores = self.model.predict(dmatrix)

        # 排序
        for item, score in zip(candidates, scores):
            item.coarse_score = score

        ranked = sorted(candidates, key=lambda x: x.coarse_score, reverse=True)

        return ranked[:top_k]

    def extract_features(self, item: RecallItem,
                         query_analysis: QueryAnalysis) -> List[float]:
        """提取粗排特征"""
        features = []

        # 文本匹配特征
        features.append(item.score)  # BM25 分数

        # 商品统计特征
        features.append(item.source.get('sales', 0))
        features.append(item.source.get('rating', 0))
        features.append(item.source.get('price', 0))

        # 类目匹配
        category_match = 1.0 if item.source.get('category') in query_analysis.categories else 0.0
        features.append(category_match)

        return features


class FineRanker:
    """精排 - 深度模型"""

    def __init__(self):
        self.model = torch.jit.load('fine_ranker.pt')  # TorchScript
        self.model.eval()

    async def rank(self, candidates: List[RankedItem],
                   query_analysis: QueryAnalysis,
                   user_id: str,
                   top_k: int = 200) -> List[RankedItem]:
        """
        精排

        特点:
        - 特征多 (数百维)
        - 模型复杂 (DeepFM/DIN)
        - 考虑用户个性化
        """
        # 获取用户画像
        user_profile = await self.get_user_profile(user_id)

        # 批量预测
        batch_size = 32
        all_scores = []

        for i in range(0, len(candidates), batch_size):
            batch = candidates[i:i+batch_size]

            # 构建特征
            features = self.build_features(batch, query_analysis, user_profile)

            # 预测
            with torch.no_grad():
                scores = self.model(features).cpu().numpy()
                all_scores.extend(scores)

        # 更新分数
        for item, score in zip(candidates, all_scores):
            item.fine_score = score

        # 排序

class ReRanker:
    """重排 - 业务规则 + 多样性"""

    def rerank(self, candidates: List[RankedItem],
               query_analysis: QueryAnalysis,
               top_k: int = 20) -> List[RankedItem]:
        """
        重排

        考虑因素:
        - 多样性 (MMR)
        - 新鲜度
        - 业务规则 (推广、过滤)
        - 疲劳度
        """
        # 1. 多样性控制
        diverse_results = self.apply_diversity(candidates, top_k)

        # 2. 业务规则
        final_results = self.apply_business_rules(diverse_results)

        return final_results[:top_k]

    def apply_diversity(self, candidates: List[RankedItem],
                        top_k: int) -> List[RankedItem]:
        """多样性重排 - MMR"""
        selected = []
        remaining = candidates[:]

        lambda_param = 0.5

        while len(selected) < top_k and remaining:
            if not selected:
                # 第一个选分数最高的
                best = remaining[0]
            else:
                # 选 MMR 分数最高的
                def mmr_score(item):
                    rel = item.fine_score

                    # 计算与已选的最大相似度
                    max_sim = max(
                        self.similarity(item, s) for s in selected
                    )

                    return lambda_param * rel - (1 - lambda_param) * max_sim

                best = max(remaining, key=mmr_score)

            selected.append(best)
            remaining.remove(best)

        return selected

  • Query 理解: 分词、纠错、实体识别、意图分类
  • 召回层: 多路召回 (文本、向量、类目、个性化)
  • 粗排: 轻量模型 (XGBoost)，快速筛选
  • 精排: 深度模型，考虑个性化
  • 重排: 多样性、业务规则

Q2: 为什么需要三级排序？

A:

  • 计算效率: 精排模型复杂，不能对所有候选排序
  • 延迟要求: 粗排快，精排慢，平衡效果和性能
  • 资源限制: GPU 资源有限，只能精排少量候选

Q3: 多路召回如何合并？

A:

  • 加权融合: 不同召回路给不同权重
  • 轮询排序: 轮流从各路取一个
  • 去重: 相同结果合并，分数累加

Q4: 如何处理 Query 纠错？

A:

  • 编辑距离: Levenshtein Distance
  • 拼音匹配: 针对拼音输入错误
  • 统计模型: 基于用户行为数据训练
  • 混合策略: 规则 + 模型

Q5: 重排阶段有哪些考虑？

A:

  • 多样性: MMR 算法避免结果同质化
  • 新鲜度: 新品加权
  • 疲劳度: 避免重复展示
  • 业务规则: 推广、过滤
