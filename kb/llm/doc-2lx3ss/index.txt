from bs4 import BeautifulSoup

def extract_text_from_html(html):
    """从 HTML 提取干净文本"""
    soup = BeautifulSoup(html, 'html.parser')

    # 移除脚本和样式
    for script in soup(['script', 'style', 'nav', 'footer']):
        script.decompose()

    # 提取文本
    text = soup.get_text()

    # 清理
    lines = (line.strip() for line in text.splitlines())
    chunks = (phrase.strip() for line in lines for phrase in line.split("  "))
    text = '\n'.join(chunk for chunk in chunks if chunk)

    return text
```

## 去重技术

### 1. 精确去重

**基于哈希**:

```python
import hashlib

def exact_deduplication(documents):
    """精确去重"""
    seen = set()
    unique_docs = []

    for doc in documents:
        doc_hash = hashlib.md5(doc.encode()).hexdigest()
        if doc_hash not in seen:
            seen.add(doc_hash)
            unique_docs.append(doc)

    return unique_docs
```

### 2. 模糊去重 (MinHash + LSH)

**MinHash**: 估计文档间的 Jaccard 相似度

```python
from datasketch import MinHash, MinHashLSH

def create_minhash(text, num_perm=128):
    """为文档创建 MinHash 签名"""
    m = MinHash(num_perm=num_perm)

    # 使用 shingle (n-gram)
    shingles = set()
    words = text.split()
    for i in range(len(words) - 2):
        shingle = ' '.join(words[i:i+3])  # 3-gram
        shingles.add(shingle)

    for s in shingles:
        m.update(s.encode('utf8'))

    return m

# LSH 索引
lsh = MinHashLSH(threshold=0.8, num_perm=128)

for i, doc in enumerate(documents):
    m = create_minhash(doc)

    # 查询相似文档
    result = lsh.query(m)
    if len(result) == 0:
        # 无相似文档，插入
        lsh.insert(f"doc_{i}", m)
        yield doc
```

### 3. 子串去重 (Suffix Array)

**去除训练数据中的测试集污染**:

```python
def remove_contamination(text, test_set, min_match_length=100):
    """移除与测试集匹配的子串"""
    for test_example in test_set:
        if test_example in text:
            text = text.replace(test_example, '')
    return text
```

### 4. 文档内去重

```python
def dedup_paragraphs(text, min_paragraph_length=50):
    """文档内段落去重"""
    paragraphs = text.split('\n\n')
    seen = set()
    unique_paragraphs = []

    for para in paragraphs:
        para = para.strip()
        if len(para) < min_paragraph_length:
            continue

        para_hash = hashlib.md5(para.lower().encode()).hexdigest()
        if para_hash not in seen:
            seen.add(para_hash)
            unique_paragraphs.append(para)

    return '\n\n'.join(unique_paragraphs)
```

## 质量评估与过滤

### 1. 启发式规则

```python
def heuristic_quality_filter(text):
    """启发式质量过滤"""
    # 1. 单词平均长度检查 (过滤随机字符)
    words = text.split()
    avg_word_len = sum(len(w) for w in words) / len(words)
    if avg_word_len > 15 or avg_word_len < 2:
        return False

    # 2. 符号比例检查
    symbol_ratio = sum(1 for c in text if not c.isalnum()) / len(text)
    if symbol_ratio > 0.5:
        return False

    # 3. 行尾省略号比例 (过滤爬虫截断)
    ellipsis_ratio = text.count('...') / len(text.split('\n'))
    if ellipsis_ratio > 0.3:
        return False

    # 4. 停用词比例 (过滤非自然语言)
    stopwords = set(['the', 'a', 'an', 'is', 'are', ...])
    stopword_count = sum(1 for w in words if w.lower() in stopwords)
    if stopword_count / len(words) < 0.1:
        return False

    return True
```

```python
def perplexity_filter(text, model, tokenizer, threshold=1000):
    """使用语言模型困惑度过滤"""
    inputs = tokenizer(text, return_tensors="pt")

    with torch.no_grad():
        outputs = model(**inputs, labels=inputs["input_ids"])
        perplexity = torch.exp(outputs.loss).item()

    return perplexity < threshold
```

### 温度采样

**调整不同领域的采样概率**:

```python
def temperature_sampling(dataset_probs, temperature=0.3):
    """
    temperature < 1: 更均衡
    temperature > 1: 更倾斜
    """
    # 应用温度
    adjusted_probs = {k: v ** (1/temperature) for k, v in dataset_probs.items()}

    # 归一化
    total = sum(adjusted_probs.values())
    normalized_probs = {k: v/total for k, v in adjusted_probs.items()}

    return normalized_probs

# 示例
probs = {'web': 0.7, 'code': 0.2, 'books': 0.1}
adjusted = temperature_sampling(probs, temperature=0.3)
# 结果: web: 0.52, code: 0.31, books: 0.17 (更均衡)
```

### 动态配比

```python
class DynamicDataMixer:
    """根据训练损失动态调整数据配比"""

    def __init__(self, datasets, initial_probs):
        self.datasets = datasets
        self.probs = initial_probs
        self.domain_losses = {k: [] for k in datasets}

    def update_probs(self, domain_losses):
        """根据领域损失调整采样概率"""
        # 损失高的领域增加采样
        avg_losses = {k: sum(v)/len(v) for k, v in domain_losses.items()}
        total_loss = sum(avg_losses.values())

        # 损失占比越高，采样概率越高
        new_probs = {k: v/total_loss for k, v in avg_losses.items()}

        # 平滑更新
        for k in self.probs:
            self.probs[k] = 0.9 * self.probs[k] + 0.1 * new_probs[k]

    def sample_batch(self, batch_size):
        """根据当前概率采样批次"""
        domains = list(self.probs.keys())
        probs = list(self.probs.values())

        selected_domains = np.random.choice(domains, size=batch_size, p=probs)

        batch = []
        for domain in selected_domains:
            sample = self.datasets[domain].sample()
            batch.append(sample)

        return batch
```

from bs4 import BeautifulSoup

def extract_text_from_html(html):
    """从 HTML 提取干净文本"""
    soup = BeautifulSoup(html, 'html.parser')

    # 移除脚本和样式
    for script in soup(['script', 'style', 'nav', 'footer']):

    # 提取文本
    text = soup.get_text()

    lines = (line.strip() for line in text.splitlines())

## 去重技术

### 1. 精确去重

**基于哈希**:

```python
import hashlib

def exact_deduplication(documents):
    """精确去重"""
    seen = set()
    unique_docs = []

    for doc in documents:
        doc_hash = hashlib.md5(doc.encode()).hexdigest()
        if doc_hash not in seen:
            seen.add(doc_hash)
            unique_docs.append(doc)

    return unique_docs
```

### 2. 模糊去重 (MinHash + LSH)

**MinHash**: 估计文档间的 Jaccard 相似度

```python
from datasketch import MinHash, MinHashLSH

def create_minhash(text, num_perm=128):
    """为文档创建 MinHash 签名"""
    m = MinHash(num_perm=num_perm)

    # 使用 shingle (n-gram)
    shingles = set()
    words = text.split()
    for i in range(len(words) - 2):
        shingle = ' '.join(words[i:i+3])  # 3-gram
        shingles.add(shingle)

    for s in shingles:
        m.update(s.encode('utf8'))

    return m

# LSH 索引
lsh = MinHashLSH(threshold=0.8, num_perm=128)

for i, doc in enumerate(documents):
    m = create_minhash(doc)

    # 查询相似文档
    result = lsh.query(m)
    if len(result) == 0:
        # 无相似文档，插入
        lsh.insert(f"doc_{i}", m)
        yield doc
```

### 3. 子串去重 (Suffix Array)

**去除训练数据中的测试集污染**:

```python
def remove_contamination(text, test_set, min_match_length=100):
    """移除与测试集匹配的子串"""
    for test_example in test_set:
        if test_example in text:
            text = text.replace(test_example, '')
    return text
```

### 4. 文档内去重

```python
def dedup_paragraphs(text, min_paragraph_length=50):
    """文档内段落去重"""
    paragraphs = text.split('\n\n')
    seen = set()
    unique_paragraphs = []

    for para in paragraphs:
        para = para.strip()
        if len(para) < min_paragraph_length:
            continue

        para_hash = hashlib.md5(para.lower().encode()).hexdigest()
        if para_hash not in seen:
            seen.add(para_hash)
            unique_paragraphs.append(para)

    return '\n\n'.join(unique_paragraphs)
```

## 质量评估与过滤

### 1. 启发式规则

```python
def heuristic_quality_filter(text):
    """启发式质量过滤"""
    # 1. 单词平均长度检查 (过滤随机字符)
    words = text.split()
    avg_word_len = sum(len(w) for w in words) / len(words)
    if avg_word_len > 15 or avg_word_len < 2:
        return False

    # 2. 符号比例检查
    symbol_ratio = sum(1 for c in text if not c.isalnum()) / len(text)
    if symbol_ratio > 0.5:
        return False

    # 3. 行尾省略号比例 (过滤爬虫截断)
    ellipsis_ratio = text.count('...') / len(text.split('\n'))
    if ellipsis_ratio > 0.3:
        return False

    # 4. 停用词比例 (过滤非自然语言)
    stopwords = set(['the', 'a', 'an', 'is', 'are', ...])
    stopword_count = sum(1 for w in words if w.lower() in stopwords)
    if stopword_count / len(words) < 0.1:
        return False

    return True
```

### 2. 基于模型的质量评分

```python
# 使用小型分类器评估质量
from transformers import AutoTokenizer, AutoModelForSequenceClassification

quality_tokenizer = AutoTokenizer.from_pretrained("quality-classifier")

def score_quality(text):
    """使用模型评估文本质量"""
    inputs = quality_tokenizer(
        text[:1024],  # 截断
        return_tensors="pt",
        truncation=True
    )
    outputs = quality_model(**inputs)
    score = torch.softmax(outputs.logits, dim=-1)[0][1].item()
    return score

# 过滤低质量文本
if score_quality(text) > 0.5:
    keep(text)
```

### 3. 困惑度过滤

```python
def perplexity_filter(text, model, tokenizer, threshold=1000):
    """使用语言模型困惑度过滤"""
    inputs = tokenizer(text, return_tensors="pt")

    with torch.no_grad():
        outputs = model(**inputs, labels=inputs["input_ids"])
        perplexity = torch.exp(outputs.loss).item()

    return perplexity < threshold
```

### 温度采样

**调整不同领域的采样概率**:

```python
def temperature_sampling(dataset_probs, temperature=0.3):
    """
    temperature < 1: 更均衡
    temperature > 1: 更倾斜
    """
    # 应用温度
    adjusted_probs = {k: v ** (1/temperature) for k, v in dataset_probs.items()}

    # 归一化
    total = sum(adjusted_probs.values())
    normalized_probs = {k: v/total for k, v in adjusted_probs.items()}

    return normalized_probs

# 示例
probs = {'web': 0.7, 'code': 0.2, 'books': 0.1}
adjusted = temperature_sampling(probs, temperature=0.3)
# 结果: web: 0.52, code: 0.31, books: 0.17 (更均衡)
```

### 动态配比

```python
class DynamicDataMixer:
    """根据训练损失动态调整数据配比"""

    def __init__(self, datasets, initial_probs):
        self.datasets = datasets
        self.probs = initial_probs
        self.domain_losses = {k: [] for k in datasets}

    def update_probs(self, domain_losses):
        """根据领域损失调整采样概率"""
        # 损失高的领域增加采样
        avg_losses = {k: sum(v)/len(v) for k, v in domain_losses.items()}
        total_loss = sum(avg_losses.values())

        # 损失占比越高，采样概率越高
        new_probs = {k: v/total_loss for k, v in avg_losses.items()}

        # 平滑更新
        for k in self.probs:
            self.probs[k] = 0.9 * self.probs[k] + 0.1 * new_probs[k]

    def sample_batch(self, batch_size):
        """根据当前概率采样批次"""
        domains = list(self.probs.keys())
        probs = list(self.probs.values())

        selected_domains = np.random.choice(domains, size=batch_size, p=probs)

        batch = []
        for domain in selected_domains:
            sample = self.datasets[domain].sample()
            batch.append(sample)

        return batch
```
