class LLMServiceGateway:
    """LLM 服务网关"""

    def __init__(self):
        self.rate_limiter = TokenBucketRateLimiter(
            capacity=1000,
            refill_rate=100
        )
        self.cache = RedisCache()
        self.router = LoadBalancer(strategy='least_conn')

    async def handle_request(self, request: LLMRequest) -> LLMResponse:
        # 1. 限流检查
        if not await self.rate_limiter.allow(request.user_id):
            raise RateLimitExceeded()

        # 2. 缓存查询
        cache_key = self._generate_cache_key(request)
        cached = await self.cache.get(cache_key)
        if cached:
            return cached

        # 3. 路由到后端服务
        backend = self.router.select(request)

        # 4. 请求处理
        response = await backend.generate(request)

        # 5. 缓存结果
        await self.cache.set(cache_key, response, ttl=3600)

        return response


class vLLMServingPod:
    """基于 vLLM 的推理服务 Pod"""

    def __init__(self):
        self.engine = LLMEngine(
            model="meta-llama/Llama-2-70b",
            tensor_parallel_size=8,
            gpu_memory_utilization=0.85,
            max_num_seqs=256,
            max_model_len=4096,
            block_size=16,
            swap_space=20,
        )

    async def generate(self, request: LLMRequest) -> LLMResponse:
        request_id = str(uuid.uuid4())

        sampling_params = SamplingParams(
            temperature=request.temperature,
            top_p=request.top_p,
            max_tokens=request.max_tokens,
            stop=request.stop_sequences,
        )

        self.engine.add_request(request_id, request.prompt, sampling_params)

        output_text = ""
        while True:
            step_outputs = self.engine.step()

            for output in step_outputs:
                if output.request_id == request_id:
                    output_text = output.outputs[-1].text

                    if output.finished:
                        return LLMResponse(
                            text=output_text,
                            usage=output.usage,
                            finish_reason=output.outputs[-1].finish_reason
                        )

            await asyncio.sleep(0.01)

class RAGSystem:
    """企业级 RAG 系统"""

    def __init__(self):
        # 向量数据库
        self.vector_store = MilvusClient(
            collection_name="documents",
            embedding_dim=1536,
            index_type="HNSW",
            metric_type="COSINE"
        )

        # 关键词搜索
        self.keyword_search = ElasticsearchClient(
            index_name="documents",
            similarity="BM25"
        )

        # 重排序模型
        self.reranker = CrossEncoderReranker()

        # LLM
        self.llm = OpenAIClient()

    async def retrieve(self, query: str, top_k: int = 20) -> List[DocumentChunk]:
        """多路召回"""
        # 查询改写
        rewritten_queries = await self.rewrite_query(query)

        results = []

        # 1. 向量召回
        query_embedding = await self.embed(rewritten_queries[0])
        vector_results = await self.vector_store.search(
            query_embedding, top_k=top_k
        )
        results.extend(vector_results)

        # 2. 关键词召回
        keyword_results = await self.keyword_search.search(
            rewritten_queries[0], top_k=top_k
        )
        results.extend(keyword_results)

        # 3. 去重
        unique_results = self.deduplicate(results)

        # 4. 重排序
        reranked = await self.reranker.rerank(
            query, unique_results, top_n=10
        )

        return reranked

    async def rewrite_query(self, query: str) -> List[str]:
        """查询改写"""
        prompt = f"""将以下查询改写为3-5个更具体、更易检索的查询：

原查询: {query}

直接输出改写后的查询列表，每行一个。"""

        response = await self.llm.generate(prompt, temperature=0.7)
        queries = [q.strip() for q in response.split('\n') if q.strip()]
        queries.insert(0, query)

        return queries[:5]
```

class LLMAgent:
    """基于 ReAct 的 LLM Agent"""

    def __init__(self, llm, tools, memory=None):
        self.llm = llm
        self.tools = ToolRegistry(tools)
        self.memory = memory or ConversationMemory()
        self.max_iterations = 10

    async def run(self, task: str, context: dict = None) -> str:
        """执行任务"""
        self.memory.add_user_message(task)

        for iteration in range(self.max_iterations):
            # 构建 ReAct 提示词
            prompt = self._build_react_prompt(
                task=task,
                history=self.memory.get_history(),
                available_tools=self.tools.get_descriptions(),
                context=context
            )

            # 调用 LLM
            response = await self.llm.generate(prompt)

            # 解析响应
            thought, action, action_input = self._parse_response(response)
            self.memory.add_assistant_message(f"Thought: {thought}")

            # 检查是否完成
            if action == "finish":
                return action_input

            # 执行工具
            if action in self.tools:
                try:
                    observation = await self.tools.execute(action, action_input)
                    self.memory.add_tool_result(action, action_input, observation)
                except Exception as e:
                    observation = f"Error: {str(e)}"
                    self.memory.add_tool_result(action, action_input, observation)

        return "达到最大迭代次数，任务未能完成"

    def _build_react_prompt(self, task, history, available_tools, context):
        """构建 ReAct 风格的提示词"""
        tools_desc = "\n".join([
            f"- {name}: {desc}"
            for name, desc in available_tools.items()
        ])

        return f"""你是一个帮助用户解决问题的 AI 助手。

可用工具:
{tools_desc}

使用以下格式回答:
Thought: 你的思考过程
Action: 你要使用的工具名称（或 finish 表示完成）
Action Input: 工具的输入参数

历史记录:
{history}

用户任务: {task}

请开始:
Thought: """

    def _parse_response(self, response: str) -> tuple:
        """解析 LLM 响应"""
        thought_match = re.search(r"Thought:\s*(.+?)(?=\nAction:|$)", response, re.DOTALL)
        action_match = re.search(r"Action:\s*(\w+)", response)
        input_match = re.search(r"Action Input:\s*(.+?)(?=\n|$)", response, re.DOTALL)

        thought = thought_match.group(1).strip() if thought_match else ""
        action = action_match.group(1).strip() if action_match else "finish"
        action_input = input_match.group(1).strip() if input_match else response

        return thought, action, action_input

class InferenceOptimizer:
    """推理优化系统"""

    def __init__(self, model_path: str, config: OptimizationConfig):

        self.model = self._load_and_optimize(model_path)

    def _load_and_optimize(self, model_path: str):

class LLMServiceGateway:
    """LLM 服务网关"""

    def __init__(self):
        self.rate_limiter = TokenBucketRateLimiter(
            capacity=1000,
            refill_rate=100
        )
        self.cache = RedisCache()
        self.router = LoadBalancer(strategy='least_conn')

    async def handle_request(self, request: LLMRequest) -> LLMResponse:
        # 1. 限流检查
        if not await self.rate_limiter.allow(request.user_id):
            raise RateLimitExceeded()

        # 2. 缓存查询
        cache_key = self._generate_cache_key(request)
        cached = await self.cache.get(cache_key)
        if cached:
            return cached

        # 3. 路由到后端服务
        backend = self.router.select(request)

        # 4. 请求处理
        response = await backend.generate(request)

        # 5. 缓存结果
        await self.cache.set(cache_key, response, ttl=3600)

        return response


class vLLMServingPod:
    """基于 vLLM 的推理服务 Pod"""

    def __init__(self):
        self.engine = LLMEngine(
            model="meta-llama/Llama-2-70b",
            tensor_parallel_size=8,
            gpu_memory_utilization=0.85,
            max_num_seqs=256,
            max_model_len=4096,
            block_size=16,
            swap_space=20,
        )

    async def generate(self, request: LLMRequest) -> LLMResponse:
        request_id = str(uuid.uuid4())

        sampling_params = SamplingParams(
            temperature=request.temperature,
            top_p=request.top_p,
            max_tokens=request.max_tokens,
            stop=request.stop_sequences,
        )

        self.engine.add_request(request_id, request.prompt, sampling_params)

        output_text = ""
        while True:
            step_outputs = self.engine.step()

            for output in step_outputs:
                if output.request_id == request_id:
                    output_text = output.outputs[-1].text

                    if output.finished:
                        return LLMResponse(
                            text=output_text,
                            usage=output.usage,
                            finish_reason=output.outputs[-1].finish_reason
                        )

            await asyncio.sleep(0.01)
```

class RAGSystem:
    """企业级 RAG 系统"""

    def __init__(self):
        # 向量数据库
        self.vector_store = MilvusClient(
            collection_name="documents",
            embedding_dim=1536,
            index_type="HNSW",
            metric_type="COSINE"
        )

        # 关键词搜索
        self.keyword_search = ElasticsearchClient(
            index_name="documents",
            similarity="BM25"
        )

        # 重排序模型
        self.reranker = CrossEncoderReranker()

        # LLM
        self.llm = OpenAIClient()

    async def retrieve(self, query: str, top_k: int = 20) -> List[DocumentChunk]:
        """多路召回"""
        # 查询改写
        rewritten_queries = await self.rewrite_query(query)

        results = []

        # 1. 向量召回
        query_embedding = await self.embed(rewritten_queries[0])
        vector_results = await self.vector_store.search(
            query_embedding, top_k=top_k
        )
        results.extend(vector_results)

        # 2. 关键词召回
        keyword_results = await self.keyword_search.search(
            rewritten_queries[0], top_k=top_k
        )
        results.extend(keyword_results)

        # 3. 去重
        unique_results = self.deduplicate(results)

        # 4. 重排序
        reranked = await self.reranker.rerank(
            query, unique_results, top_n=10
        )

        return reranked

    async def rewrite_query(self, query: str) -> List[str]:
        """查询改写"""
        prompt = f"""将以下查询改写为3-5个更具体、更易检索的查询：

原查询: {query}

直接输出改写后的查询列表，每行一个。"""

        response = await self.llm.generate(prompt, temperature=0.7)
        queries = [q.strip() for q in response.split('\n') if q.strip()]
        queries.insert(0, query)

        return queries[:5]
```

```python
class LLMAgent:
    """基于 ReAct 的 LLM Agent"""

    def __init__(self, llm, tools, memory=None):
        self.llm = llm
        self.tools = ToolRegistry(tools)
        self.memory = memory or ConversationMemory()
        self.max_iterations = 10

    async def run(self, task: str, context: dict = None) -> str:
        """执行任务"""
        self.memory.add_user_message(task)

        for iteration in range(self.max_iterations):
            # 构建 ReAct 提示词
            prompt = self._build_react_prompt(
                task=task,
                history=self.memory.get_history(),
                available_tools=self.tools.get_descriptions(),
                context=context
            )

            # 调用 LLM
            response = await self.llm.generate(prompt)

            # 解析响应
            thought, action, action_input = self._parse_response(response)
            self.memory.add_assistant_message(f"Thought: {thought}")

            # 检查是否完成
            if action == "finish":
                return action_input

            # 执行工具
            if action in self.tools:
                try:
                    observation = await self.tools.execute(action, action_input)
                    self.memory.add_tool_result(action, action_input, observation)
                except Exception as e:
                    observation = f"Error: {str(e)}"
                    self.memory.add_tool_result(action, action_input, observation)

        return "达到最大迭代次数，任务未能完成"

    def _build_react_prompt(self, task, history, available_tools, context):
        """构建 ReAct 风格的提示词"""
        tools_desc = "\n".join([
            f"- {name}: {desc}"
            for name, desc in available_tools.items()
        ])

        return f"""你是一个帮助用户解决问题的 AI 助手。

可用工具:
{tools_desc}

使用以下格式回答:
Thought: 你的思考过程
Action: 你要使用的工具名称（或 finish 表示完成）
Action Input: 工具的输入参数

历史记录:
{history}

用户任务: {task}

请开始:
Thought: """

    def _parse_response(self, response: str) -> tuple:
        """解析 LLM 响应"""
        thought_match = re.search(r"Thought:\s*(.+?)(?=\nAction:|$)", response, re.DOTALL)
        action_match = re.search(r"Action:\s*(\w+)", response)
        input_match = re.search(r"Action Input:\s*(.+?)(?=\n|$)", response, re.DOTALL)

        thought = thought_match.group(1).strip() if thought_match else ""
        action = action_match.group(1).strip() if action_match else "finish"
        action_input = input_match.group(1).strip() if input_match else response

        return thought, action, action_input
```
