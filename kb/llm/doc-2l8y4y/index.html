多模态概述

什么是多模态

多模态模型能够处理和理解多种类型的数据（文本、图像、音频、视频）。

单模态: 文本 → 文本 (GPT)
多模态: 文本 + 图像 → 文本 (GPT-4V)

应用场景:
- 图像描述生成
- 视觉问答 (VQA)
- 图文检索
- 视频理解
- 跨模态翻译

视觉编码器

Vision Transformer (ViT)

class VisionTransformer(nn.Module):
    """视觉 Transformer"""

    def __init__(self, img_size=224, patch_size=16, in_channels=3,
                 embed_dim=768, depth=12, num_heads=12, mlp_ratio=4):
        super().__init__()

        self.patch_embed = PatchEmbed(
            img_size=img_size,
            patch_size=patch_size,
            in_channels=in_channels,
            embed_dim=embed_dim
        )

        # 位置编码
        num_patches = self.patch_embed.num_patches
        self.pos_embed = nn.Parameter(torch.zeros(1, num_patches + 1, embed_dim))
        self.cls_token = nn.Parameter(torch.zeros(1, 1, embed_dim))

        # Transformer 块
        self.blocks = nn.ModuleList([
            TransformerBlock(embed_dim, num_heads, mlp_ratio)
            for _ in range(depth)
        ])

        self.norm = nn.LayerNorm(embed_dim)

    def forward(self, x):
        """
        x: [batch, 3, 224, 224]
        output: [batch, num_patches+1, embed_dim]
        """
        batch_size = x.shape[0]

        # 1. 图像分块
        x = self.patch_embed(x)  # [batch, num_patches, embed_dim]

        # 2. 添加 CLS token
        cls_tokens = self.cls_token.expand(batch_size, -1, -1)
        x = torch.cat([cls_tokens, x], dim=1)

        # 3. 添加位置编码
        x = x + self.pos_embed

        # 4. Transformer 编码
        for block in self.blocks:
            x = block(x)

        x = self.norm(x)

        # 返回 CLS token 和所有 patch tokens
        return x  # [batch, num_patches+1, embed_dim]


class PatchEmbed(nn.Module):
    """图像分块嵌入"""

    def __init__(self, img_size=224, patch_size=16, in_channels=3, embed_dim=768):
        super().__init__()

        self.img_size = img_size
        self.patch_size = patch_size
        self.num_patches = (img_size // patch_size) ** 2

        self.proj = nn.Conv2d(
            in_channels, embed_dim,
            kernel_size=patch_size,
            stride=patch_size
        )

    def forward(self, x):
        """
        x: [batch, 3, 224, 224]
        output: [batch, num_patches, embed_dim]
        """
        x = self.proj(x)  # [batch, embed_dim, 14, 14]
        x = x.flatten(2)  # [batch, embed_dim, 196]
        x = x.transpose(1, 2)  # [batch, 196, embed_dim]
        return x

Swin Transformer

class SwinTransformerBlock(nn.Module):
    """Swin Transformer 块 - 使用局部窗口注意力"""

    def __init__(self, dim, num_heads, window_size=7, shift_size=0):
        super().__init__()

        self.dim = dim
        self.num_heads = num_heads
        self.window_size = window_size
        self.shift_size = shift_size

        self.norm1 = nn.LayerNorm(dim)
        self.attn = WindowAttention(
            dim, num_heads, window_size=window_size
        )

        self.norm2 = nn.LayerNorm(dim)
        self.mlp = MLP(dim, int(dim * 4))

    def forward(self, x, H, W):
        """
        x: [batch, H*W, dim]
        """
        shortcut = x
        x = self.norm1(x)

        # reshape 为 2D
        x = x.view(-1, H, W, self.dim)

        # cyclic shift
        if self.shift_size > 0:
            shifted_x = torch.roll(
                x, shifts=(-self.shift_size, -self.shift_size), dims=(1, 2)
            )
        else:
            shifted_x = x

        # 划分窗口
        x_windows = self.window_partition(shifted_x, self.window_size)
        # x_windows: [num_windows*batch, window_size*window_size, dim]

        # 窗口注意力
        attn_windows = self.attn(x_windows)

        # 合并窗口
        shifted_x = self.window_reverse(attn_windows, self.window_size, H, W)

        # reverse cyclic shift
        if self.shift_size > 0:
            x = torch.roll(
                shifted_x, shifts=(self.shift_size, self.shift_size), dims=(1, 2)
            )
        else:
            x = shifted_x

        # reshape 回 1D
        x = x.view(-1, H * W, self.dim)

        # FFN
        x = shortcut + x
        x = x + self.mlp(self.norm2(x))

        return x

多模态融合

早期融合 (Early Fusion)

class EarlyFusion(nn.Module):
    """早期融合: 在特征层面融合"""

    def __init__(self, text_encoder, vision_encoder, fusion_dim):
        super().__init__()

        self.text_encoder = text_encoder
        self.vision_encoder = vision_encoder

        # 投影到统一空间
        self.text_proj = nn.Linear(text_encoder.config.hidden_size, fusion_dim)
        self.vision_proj = nn.Linear(vision_encoder.config.hidden_size, fusion_dim)

        # 融合层
        self.fusion_layer = nn.TransformerEncoderLayer(
            d_model=fusion_dim,
            nhead=8,
            dim_feedforward=fusion_dim * 4
        )

    def forward(self, images, text_tokens):
        # 编码
        vision_features = self.vision_encoder(images)  # [batch, num_patches+1, v_dim]
        text_features = self.text_encoder(text_tokens)  # [batch, seq_len, t_dim]

        # 投影
        vision_features = self.vision_proj(vision_features)
        text_features = self.text_proj(text_features)

        # 拼接并融合
        combined = torch.cat([vision_features, text_features], dim=1)
        fused = self.fusion_layer(combined)

        return fused

晚期融合 (Late Fusion)

class LateFusion(nn.Module):
    """晚期融合: 分别编码后在高层融合"""

    def __init__(self, text_encoder, vision_encoder, fusion_dim):
        super().__init__()

        self.text_encoder = text_encoder
        self.vision_encoder = vision_encoder

        # 投影到统一空间
        self.text_proj = nn.Linear(text_encoder.config.hidden_size, fusion_dim)
        self.vision_proj = nn.Linear(vision_encoder.config.hidden_size, fusion_dim)

        # 融合层
        self.fusion_mlp = nn.Sequential(
            nn.Linear(fusion_dim * 2, fusion_dim),
            nn.ReLU(),
            nn.Linear(fusion_dim, fusion_dim)
        )

    def forward(self, images, text_tokens):
        # 分别编码
        vision_features = self.vision_encoder(images)
        text_features = self.text_encoder(text_tokens)

        # 取全局特征 (CLS token 或平均)
        vision_global = vision_features[:, 0]  # CLS token
        text_global = text_features[:, 0]  # [CLS] token

        # 投影
        vision_proj = self.vision_proj(vision_global)
        text_proj = self.text_proj(text_global)

        # 拼接并融合
        combined = torch.cat([vision_proj, text_proj], dim=-1)
        fused = self.fusion_mlp(combined)

        return fused

门控融合

class GatedFusion(nn.Module):
    """门控多模态融合"""

    def __init__(self, dim):
        super().__init__()

        # 门控网络
        self.gate = nn.Sequential(
            nn.Linear(dim * 2, dim),
            nn.Sigmoid()
        )

        self.text_transform = nn.Linear(dim, dim)
        self.vision_transform = nn.Linear(dim, dim)

    def forward(self, text_features, vision_features):
        """
        text_features: [batch, seq_len, dim]
        vision_features: [batch, num_patches, dim]
        """
        # 平均池化得到全局特征
        text_global = text_features.mean(dim=1)
        vision_global = vision_features.mean(dim=1)

        # 计算门控权重
        gate_input = torch.cat([text_global, vision_global], dim=-1)
        gate = self.gate(gate_input)  # [batch, dim]

        # 门控融合
        text_trans = self.text_transform(text_global)
        vision_trans = self.vision_transform(vision_global)

        fused = gate * text_trans + (1 - gate) * vision_trans

        return fused

CLIP 与对比学习

CLIP 架构

class CLIP(nn.Module):
    """
    CLIP: Contrastive Language-Image Pre-training

    核心思想: 使用对比学习对齐图文特征空间
    """

    def __init__(self, text_encoder, vision_encoder, embed_dim=512):
        super().__init__()

        self.text_encoder = text_encoder
        self.vision_encoder = vision_encoder

        # 投影头
        self.text_projection = nn.Linear(
            text_encoder.config.hidden_size, embed_dim
        )
        self.vision_projection = nn.Linear(
            vision_encoder.config.hidden_size, embed_dim
        )

        self.logit_scale = nn.Parameter(torch.ones([]) * np.log(1 / 0.07))

    def forward(self, images, text_tokens):
        """
        images: [batch, 3, 224, 224]
        text_tokens: [batch, seq_len]

        Returns:
            logits_per_image: [batch, batch]
            logits_per_text: [batch, batch]
        """
        # 编码
        image_features = self.vision_encoder(images)
        text_features = self.text_encoder(text_tokens)

        # 取 CLS token
        image_features = image_features[:, 0]
        text_features = text_features[:, 0]

        # 投影
        image_features = self.vision_projection(image_features)
        text_features = self.text_projection(text_features)

        # L2 归一化
        image_features = F.normalize(image_features, dim=-1)
        text_features = F.normalize(text_features, dim=-1)

        # 计算相似度
        logit_scale = self.logit_scale.exp()
        logits_per_image = logit_scale * image_features @ text_features.T
        logits_per_text = logits_per_image.T

        return logits_per_image, logits_per_text

    def contrastive_loss(self, logits_per_image, logits_per_text):
        """对比损失"""
        batch_size = logits_per_image.shape[0]

        # 对角线为正样本
        labels = torch.arange(batch_size, device=logits_per_image.device)

        # 对称损失
        loss_i2t = F.cross_entropy(logits_per_image, labels)
        loss_t2i = F.cross_entropy(logits_per_text, labels)

        loss = (loss_i2t + loss_t2i) / 2

        return loss

应用

class CLIPApplications:
    """CLIP 应用示例"""

    def __init__(self, clip_model):
        self.clip = clip_model

    def image_text_retrieval(self, images, texts, k=5):
        """图文检索"""

        # 编码所有图像
        image_features = []
        for img in images:
            feat = self.clip.encode_image(img)
            image_features.append(feat)
        image_features = torch.stack(image_features)

        # 编码所有文本
        text_features = []
        for text in texts:
            feat = self.clip.encode_text(text)
            text_features.append(feat)
        text_features = torch.stack(text_features)

        # 计算相似度
        similarity = image_features @ text_features.T

        # Top-K 检索
        top_k_indices = similarity.topk(k, dim=1).indices

        return top_k_indices

    def zero_shot_classification(self, image, class_names):
        """零样本分类"""

        # 构建 prompts
        templates = [
            "a photo of a {}",
            "a blurry photo of a {}",
            "a photo of many {}",
        ]

        # 编码所有类别的文本
        text_features = []
        for class_name in class_names:
            # 使用多个模板取平均
            texts = [template.format(class_name) for template in templates]
            text_feats = [self.clip.encode_text(t) for t in texts]
            text_features.append(torch.stack(text_feats).mean(dim=0))

        text_features = torch.stack(text_features)

        # 编码图像
        image_feature = self.clip.encode_image(image)

        # 计算相似度
        similarity = image_feature @ text_features.T

        # 预测
        predicted_class = similarity.argmax(dim=-1)

        return predicted_class

视觉语言模型

LLaVA 架构

class LLaVA(nn.Module):
    """
    LLaVA: Large Language and Vision Assistant

    架构: 视觉编码器 + 投影层 + 冻结 LLM
    """

    def __init__(self, vision_encoder, llm, projection_dim=4096):
        super().__init__()

        # 视觉编码器 (冻结)
        self.vision_encoder = vision_encoder
        for param in self.vision_encoder.parameters():
            param.requires_grad = False

        # 投影层 (可训练)
        self.mm_projector = nn.Sequential(
            nn.Linear(vision_encoder.config.hidden_size, projection_dim),
            nn.GELU(),
            nn.Linear(projection_dim, llm.config.hidden_size)
        )

        # 大语言模型 (冻结或 LoRA)
        self.llm = llm

    def forward(self, images, text_tokens):
        """
        images: [batch, 3, 336, 336]
        text_tokens: [batch, seq_len]
        """
        # 1. 视觉编码
        with torch.no_grad():
            vision_features = self.vision_encoder(images)

        # 2. 投影到 LLM 空间
        projected_features = self.mm_projector(vision_features)

        # 3. 文本编码
        text_embeds = self.llm.get_input_embeddings()(text_tokens)

        # 4. 拼接
        # 找到 <image> token 的位置，用视觉特征替换
        combined_embeds = self.insert_vision_tokens(
            text_embeds, projected_features
        )

        # 5. LLM 生成
        outputs = self.llm(inputs_embeds=combined_embeds)

        return outputs

    def generate(self, image, prompt, max_new_tokens=512):
        """生成回答"""

        # 构建对话格式
        full_prompt = f"USER: <image>\n{prompt}\nASSISTANT:"

        # 编码
        inputs = self.tokenizer(full_prompt, return_tensors="pt")

        # 生成
        outputs = self.llm.generate(
            **inputs,
            max_new_tokens=max_new_tokens,
            temperature=0.2,
            do_sample=True
        )

        response = self.tokenizer.decode(outputs[0], skip_special_tokens=True)

        # 提取 ASSISTANT 的回答
        return response.split("ASSISTANT:")[-1].strip()

Q-Former (BLIP-2)

class QFormer(nn.Module):
    """
    Q-Former: 查询 Transformer，用于压缩视觉特征

    BLIP-2 中的关键组件
    """

    def __init__(self, num_query_tokens=32, encoder_width=1408,
                 hidden_size=768, num_hidden_layers=12):
        super().__init__()

        # 可学习的查询 token
        self.query_tokens = nn.Parameter(
            torch.zeros(1, num_query_tokens, hidden_size)
        )

        # Q-Former (双向 Transformer)
        self.qformer = BertLMHeadModel(
            config=BertConfig(
                hidden_size=hidden_size,
                num_hidden_layers=num_hidden_layers,
                is_decoder=False  # 双向注意力
            )
        )

        # 投影到 LLM 维度
        self.llm_proj = nn.Linear(hidden_size, 4096)

    def forward(self, image_embeds, text_tokens=None):
        """
        image_embeds: [batch, num_patches, encoder_width]
        text_tokens: [batch, text_len] (可选)

        Returns:
            query_output: [batch, num_query_tokens, llm_dim]
        """
        batch_size = image_embeds.shape[0]

        # 扩展查询 token
        query_tokens = self.query_tokens.expand(batch_size, -1, -1)

        # 如果提供了文本，拼接
        if text_tokens is not None:
            text_embeds = self.qformer.embeddings(text_tokens)
            input_embeds = torch.cat([query_tokens, text_embeds], dim=1)
        else:
            input_embeds = query_tokens

        # 构建 attention mask
        # 查询 token 可以看到图像特征，但看不到彼此（防止信息泄露）
        # 这里简化为标准 attention

        # 通过 Q-Former
        outputs = self.qformer(
            inputs_embeds=input_embeds,
            encoder_hidden_states=image_embeds,
            return_dict=True
        )

        # 提取查询 token 的输出
        query_output = outputs.last_hidden_state[:, :self.num_query_tokens]

        # 投影到 LLM 维度
        query_output = self.llm_proj(query_output)

        return query_output

Q3: LLaVA 和 BLIP-2 的区别？

A:

维度LLaVABLIP-2
视觉编码CLIP ViTEVA-CLIP
连接方式简单 MLPQ-Former
训练策略端到端分阶段

Q4: 如何实现零样本分类？

A:

  1. 将类别名转换为文本描述（如 "a photo of a dog"）
  2. 使用 CLIP 分别编码图像和所有类别的文本
  3. 计算图像与各类别的相似度
  4. 选择相似度最高的类别作为预测结果
