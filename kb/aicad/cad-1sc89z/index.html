2. Service 层

SandboxMCPService 类

# sandbox_mcp_server/service.py

from typing import Optional
from pydantic import BaseModel

class SandboxMCPService:
    """Sandbox MCP 核心服务类"""

    def __init__(
        self,
        sandbox_runner: DockerSandboxRunner,
        llm_client: LLMClient,
        llm_judge_enabled: bool = False,
        benchmark_index_path: Optional[str] = None
    ):
        self.sandbox_runner = sandbox_runner
        self.llm_client = llm_client
        self.llm_judge_enabled = llm_judge_enabled
        self.benchmark_index_path = benchmark_index_path

    async def execute_cadquery(
        self,
        code: str,
        timeout_seconds: int = 120,
        include_artifact_content: bool = True,
        benchmark_sample_id: Optional[str] = None,
        requirement_text: Optional[str] = None
    ) -> CadQueryResult:
        """执行 CadQuery 代码"""
        # 1. 在 sandbox 中执行代码
        execution_result = await self.sandbox_runner.execute(
            code=code,
            timeout_seconds=timeout_seconds
        )

        # 2. 处理执行失败
        if not execution_result.success:
            return CadQueryResult(
                success=False,
                error_code=execution_result.error_code,
                error_message=execution_result.error_message
            )

        # 3. 构建证据
        evidence = self._build_evidence(
            code=code,
            execution_result=execution_result,
            requirement_text=requirement_text
        )

        # 4. 执行评估
        if self.llm_judge_enabled:
            evaluation = await self._evaluate_with_llm_judge(evidence)
        elif benchmark_sample_id:
            evaluation = await self._evaluate_with_ground_truth(
                execution_result,
                benchmark_sample_id
            )
        else:
            evaluation = None

        # 5. 返回结果
        return CadQueryResult(
            success=True,
            output_files=execution_result.output_files,
            artifacts=self._process_artifacts(
                execution_result.artifacts,
                include_artifact_content
            ),
            evaluation=evaluation
        )

3. Evidence Builder

# sandbox_mcp_server/evidence_builder.py

class EvidenceBuilder:
    """证据构建器 - 收集评估所需的所有信息"""

    def build(
        self,
        code: str,
        execution_result: SandboxExecutionResult,
        requirement_text: Optional[str] = None
    ) -> EvidenceBundle:
        """构建完整的证据包"""

        return EvidenceBundle(
            request_context=self._build_request_context(code, requirement_text),
            execution_context=self._build_execution_context(execution_result),
            artifact_context=self._build_artifact_context(execution_result),
            code_context=self._build_code_context(code),
            meta=self._build_meta(),
            trace=self._build_trace()
        )

    def _build_request_context(self, code: str, requirement_text: str) -> RequestContext:
        """构建请求上下文"""
        return RequestContext(
            user_prompt=requirement_text or "",
            user_code=code
        )

    def _build_execution_context(self, result: SandboxExecutionResult) -> ExecutionContext:
        """构建执行上下文"""
        return ExecutionContext(
            success=result.success,
            error_code=result.error_code,
            stderr=result.stderr,
            timeout=result.timeout
        )

    def _build_artifact_context(self, result: SandboxExecutionResult) -> ArtifactContext:
        """构建产物上下文"""
        has_step = "model.step" in result.output_files
        return ArtifactContext(
            has_step=has_step,
            step_filename="model.step" if has_step else None,
            step_size_bytes=self._get_file_size(result.output_files, "model.step"),
            has_preview=len(result.artifacts) > 0
        )

4. Rule Evaluator

# sandbox_mcp_server/rule_evaluator.py

class RuleEvaluator:
    """规则评估器 - 硬性规则检查"""

    RULES = {
        "missing_step_max_score": 0.20,
        "empty_artifact_max_score": 0.30,
        "timeout_max_score": 0.10,
        "execution_error_max_score": 0.0
    }

    def evaluate(self, evidence: EvidenceBundle) -> RuleResult:
        """执行规则评估"""

        score_cap = 1.0  # 默认无上限

        # 规则 1: 无 STEP 产物
        if not evidence.artifact_context.has_step:
            score_cap = self.RULES["missing_step_max_score"]

        # 规则 2: 空文件
        elif evidence.artifact_context.step_size_bytes < 100:
            score_cap = self.RULES["empty_artifact_max_score"]

        # 规则 3: 超时
        elif evidence.execution_context.timeout:
            score_cap = self.RULES["timeout_max_score"]

        # 规则 4: 执行错误
        elif not evidence.execution_context.success:
            score_cap = self.RULES["execution_error_max_score"]

        return RuleResult(
            passed=True,
            score_cap=score_cap,
            violations=self._detect_violations(evidence)
        )

5. LLM Judge

# sandbox_mcp_server/llm_judge.py

class LLMJudgeEvaluator:
    """LLM Judge 评估器"""

    def __init__(self, llm_client: LLMClient, rubric: RubricConfig):
        self.llm_client = llm_client
        self.rubric = rubric

    async def evaluate(self, evidence: EvidenceBundle) -> LLMJudgeResult:
        """执行 LLM Judge 评估"""

        # 1. 构建 prompt
        prompt = self._build_prompt(evidence)

        # 2. 调用 LLM
        raw_response = await self.llm_client.generate_json(
            model=self.rubric.judge_model,
            messages=[{"role": "user", "content": prompt}]
        )

        # 3. Schema 校验
        validated = self._validate_schema(raw_response)
        if not validated.ok:
            # 重试一次
            raw_response = await self.llm_client.generate_json(
                model=self.rubric.judge_model,
                messages=[{"role": "user", "content": prompt + "\n\n重要：请严格按照 JSON schema 输出"}]
            )
            validated = self._validate_schema(raw_response)

        # 4. 解析结果
        result = validated.parsed

        # 5. 计算分数
        score = self._calculate_weighted_score(result.dimension_scores, self.rubric.weights)

        return LLMJudgeResult(
            status="success",
            score=score,
            confidence=result.confidence,
            dimension_scores=result.dimension_scores,
            major_issues=result.major_issues,
            suggestions=result.suggestions
        )

6. Score Merger

# sandbox_mcp_server/score_merger.py

class ScoreMerger:
    """分数合并器 - 合并规则评估和 LLM Judge 结果"""

    def merge(
        self,
        rule_result: RuleResult,
        llm_result: Optional[LLMJudgeResult],
        rubric_config: RubricConfig
    ) -> Evaluation:
        """合并评估结果"""

        # 1. 应用规则封顶
        final_score = 1.0
        if llm_result:
            final_score = min(llm_result.score, rule_result.score_cap)
        else:
            final_score = rule_result.score_cap

        # 2. 构建评估对象
        evaluation = Evaluation(
            mode="llm_judge" if llm_result else "rule_only",
            status="success" if llm_result else rule_result.to_status(),
            score=final_score,
            details=EvaluationDetails(
                rule_result=rule_result.to_dict(),
                llm_result=llm_result.to_dict() if llm_result else None,
                rubric_version=rubric_config.version
            )
        )

        return evaluation

执行流程详解

1. execute_cadquery 完整流程

async def execute_cadquery_tool(arguments):
    """execute_cadquery 工具的完整实现"""

    # 1. 参数验证
    code = arguments.get("code")
    timeout = arguments.get("timeout", 120)

    # 2. 执行代码
    execution = await sandbox_runner.execute(code, timeout)

    # 3. 收集产物
    artifacts = collect_artifacts(execution.output_directory)

    # 4. 构建证据
    evidence = evidence_builder.build(
        code=code,
        execution_result=execution,
        requirement_text=arguments.get("requirement_text")
    )

    # 5. 规则评估
    rule_result = rule_evaluator.evaluate(evidence)

    # 6. LLM Judge 评估（如果启用）
    if llm_judge_enabled:
        llm_result = await llm_judge.evaluate(evidence)
    elif benchmark_sample_id:
        llm_result = ground_truth_evaluator.evaluate(
            execution, benchmark_sample_id
        )
    else:
        llm_result = None

    # 7. 合并分数
    evaluation = score_merger.merge(rule_result, llm_result)

    # 8. 构建响应
    return ToolResponse(
        content=[TextContent(type="text", text=format_summary(evaluation))],
        artifacts=build_artifact_contents(artifacts)
    )

2. apply_cad_action 完整流程

async def apply_cad_action_tool(arguments):
    """apply_cad_action 工具的完整实现"""

    # 1. 获取或创建会话
    session_id = arguments.get("session_id", generate_id())
    session = session_manager.get_or_create_session(session_id)

    # 2. 解析动作
    action = parse_action(arguments["action"])

    # 3. 执行动作
    result = await session.execute_action(action)

    # 4. 追加到历史
    action_entry = ActionHistoryEntry(
        step=session.current_step,
        action_type=action.type,
        params=action.params,
        snapshot=result.snapshot,
        success=result.success,
        error=result.error
    )
    session_manager.append_action(session_id, action_entry)

    # 5. 生成建议和完整性
    suggestions = generate_suggestions(result.snapshot)
    completeness = generate_completeness(session, result.snapshot)

    # 6. 构建响应
    return ToolResponse(
        content=[
            TextContent(type="text", text=f"执行动作: {action.type}"),
            TextContent(type="text", text=format_suggestions(suggestions)),
            TextContent(type="text", text=format_completeness(completeness))
        ],
        action_history=session_manager.get_session_history(session_id)
    )

产物收集

def collect_artifacts(output_dir: Path) -> list[Artifact]:
    """收集所有产物"""

    artifacts = []

    # 1. STEP 文件
    step_path = output_dir / "model.step"
    if step_path.exists():
        artifacts.append(Artifact(
            type=ArtifactType.STEP_FILE,
            path=str(step_path),
            size_bytes=step_path.stat().st_size,
            mime_type="model/step"
        ))

    # 2. 预览图
    preview_pattern = "preview_*.png"
    for preview_path in output_dir.glob(preview_pattern):
        artifacts.append(Artifact(
            type=ArtifactType.PREVIEW_IMAGE,
            path=str(preview_path),
            size_bytes=preview_path.stat().st_size,
            mime_type="image/png"
        ))

    # 3. 日志文件
    log_path = output_dir / "execution.log"
    if log_path.exists():
        artifacts.append(Artifact(
            type=ArtifactType.LOG_FILE,
            path=str(log_path),
            size_bytes=log_path.stat().st_size,
            mime_type="text/plain"
        ))

    return artifacts

产物内容编码

def encode_artifact_content(file_path: Path) -> str:
    """将文件内容编码为 base64"""

    with open(file_path, "rb") as f:
        content = f.read()

    # 对于大文件，限制大小
    max_size = 6 * 1024 * 1024  # 6MB
    if len(content) > max_size:
        return None  # 不返回大文件内容

    return base64.b64encode(content).decode('utf-8')

安全机制

1. 代码注入防护

import ast
import re

def validate_code_safety(code: str) -> SafetyCheckResult:
    """代码安全验证"""

    # 1. 检查危险函数调用
    dangerous_functions = ['eval', 'exec', 'compile', 'open', 'import os', 'import subprocess']
    for func in dangerous_functions:
        if func in code:
            return SafetyCheckResult(
                safe=False,
                reason=f"包含危险函数: {func}"
            )

    # 2. 检查文件操作
    file_operations = re.findall(r'\b(open|read|write)\b\(["\'][\w+]', code)
    if file_operations:
        return SafetyCheckResult(
                safe=False,
                reason="包含文件操作"
            )

    # 3. 检查网络操作
    network_operations = re.findall(r'\b(urllib|requests|http|https)\b', code)
    if network_operations:
        return SafetyCheckResult(
                safe=False,
                reason="包含网络请求"
            )

    # 4. AST 解析检查
    try:
        tree = ast.parse(code)
        # 检查是否有可疑节点
        for node in ast.walk(tree):
            if isinstance(node, (ast.Exec, ast.Import)):
                if not isinstance(node, ast.alias):
                    return SafetyCheckResult(
                        safe=False,
                        reason=f"包含动态执行节点"
                    )
    except:
        # 语法错误会在执行时捕获
        pass

    return SafetyCheckResult(safe=True, reason=None)

2. 资源限制

class ResourceLimits:
    """资源限制配置"""

    MAX_CODE_LENGTH = 10000      # 最大代码长度
    MAX_EXECUTION_TIME = 600      # 最大执行时间（秒）
    MAX_ARTIFACT_SIZE = 10 * 1024 * 1024  # 最大产物大小 (10MB)
    MAX_PREVIEW_IMAGES = 4       # 最大预览图数量
    MAX_SESSION_HISTORY = 50      # 最大会话历史长度

    @classmethod
    def validate_code_length(cls, code: str) -> bool:
        """验证代码长度"""
        return len(code) <= cls.MAX_CODE_LENGTH

    @classmethod
    def validate_execution_time(cls, timeout: int) -> bool:
        """验证执行时间"""
        return timeout <= cls.MAX_EXECUTION_TIME

    @classmethod
    def validate_artifact_size(cls, size: int) -> bool:
        """验证产物大小"""
        return size <= cls.MAX_ARTIFACT_SIZE

3. 超时处理

import asyncio

async def execute_with_timeout(coro, timeout_seconds):
    """带超时的异步执行"""

    try:
        result = await asyncio.wait_for(
            coro,
            timeout=timeout_seconds
        )
        return result, False

    except asyncio.TimeoutError:
        return None, True
