1. Plugin System

class PluginManager:
    """жҸ’д»¶з®ЎзҗҶеҷЁ"""

    def __init__(self):
        self.plugins: dict[str, Plugin] = {}

    def register(self, plugin: Plugin):
        """жіЁеҶҢжҸ’д»¶"""
        self.plugins[plugin.name] = plugin

    def get(self, name: str) -> Plugin:
        """иҺ·еҸ–жҸ’д»¶"""
        return self.plugins.get(name)

    def execute(
        self,
        plugin_name: str,
        method: str,
        **kwargs
    ) -> PluginResult:
        """жү§иЎҢжҸ’д»¶ж–№жі•"""
        plugin = self.get(plugin_name)
        if not plugin:
            return PluginResult(success=False, error=f"жҸ’д»¶ {plugin_name} жңӘжүҫеҲ°")

        method_handler = getattr(plugin, method, None)
        if not method_handler:
            return PluginResult(success=False, error=f"ж–№жі• {method} дёҚеӯҳеңЁ")

        try:
            return await method_handler(**kwargs)
        except Exception as e:
            return PluginResult(success=False, error=str(e))

class StateManager:
    """зҠ¶жҖҒз®ЎзҗҶеҷЁ - и·ҹиёӘ Agent зҡ„жүҖжңүзҠ¶жҖҒ"""

    def __init__(self):
        self.current_state: dict = {}
        self.history: list[dict] = []
        self.variables: dict = {}

    def update_state(self, key: str, value: any):
        """жӣҙж–°зҠ¶жҖҒ"""
        old_value = self.current_state.get(key)
        self.current_state[key] = value
        self.history.append({
            "key": key,
            "old_value": old_value,
            "new_value": value,
            "timestamp": datetime.now().isoformat()
        })

    def get_state(self, key: str) -> any:
        """иҺ·еҸ–зҠ¶жҖҒ"""
        return self.current_state.get(key)

    def get_recent_history(self, key: str, limit: int = 10) -> list[dict]:
        """иҺ·еҸ–зҠ¶жҖҒеҺҶеҸІ"""
        return [h for h in self.history if h["key"] == key][-limit:]

    def rollback_to(self, snapshot_id: str) -> bool:
        """еӣһйҖҖеҲ°жҢҮе®ҡеҝ«з…§"""
        snapshot = self.get_snapshot(snapshot_id)
        if not snapshot:
            return False

        self.current_state = snapshot["state"].copy()
        self.history = snapshot["history"].copy()
        return True

class Orchestrator:
    """зј–жҺ’еҷЁ - еҚҸи°ғеҗ„дёӘз»„д»¶"""

    def __init__(
        self,
        plugin_manager: PluginManager,
        state_manager: StateManager,
        llm_client: LLMClient
    ):
        self.plugin_manager = plugin_manager
        self.state_manager = state_manager
        self.llm_client = llm_client

    async def execute_task(self, task: Task) -> TaskResult:
        """жү§иЎҢд»»еҠЎ"""
        # 1. и§„еҲ’
        plan = await self._plan_task(task)

        # 2. жү§иЎҢ
        results = []
        for step in plan.steps:
            result = await self._execute_step(step)
            results.append(result)

        # 3. йӘҢиҜҒ
        validation = await self._validate_results(results)

        return TaskResult(
            success=validation.passed,
            results=results,
            plan=plan
        )

    async def _plan_task(self, task: Task) -> ExecutionPlan:
        """и§„еҲ’д»»еҠЎжӯҘйӘӨ"""
        # дҪҝз”Ё LLM еҲҶжһҗд»»еҠЎ
        prompt = self._build_planning_prompt(task)

        response = await self.llm_client.generate_json(prompt)

        plan = ExecutionPlan.parse(response)

        return plan

    async def _execute_step(self, step: Step) -> StepResult:
        """жү§иЎҢеҚ•дёӘжӯҘйӘӨ"""
        plugin = self.plugin_manager.get(step.plugin_name)

        # жү§иЎҢжҸ’д»¶ж–№жі•
        result = await plugin.execute(step.method, **step.parameters)

        # жӣҙж–°зҠ¶жҖҒ
        self.state_manager.update_state(step.plugin_name, result.state)

        return StepResult(success=result.success, output=result.output)

class CoreAgent:
    """ж ёеҝғд»ЈзҗҶ - еӨ„зҗҶйҖҡз”Ёд»»еҠЎ"""

    def __init__(self):
        self.capabilities = {
            "code_editing": True,
            "file_operations": True,
            "web_search": True,
            "terminal": True
        }

    async def process(self, user_message: str) -> AgentResponse:
        """еӨ„зҗҶз”ЁжҲ·ж¶ҲжҒҜ"""

        # 1. еҲҶжһҗж„Ҹеӣҫ
        intent = await self.detect_intent(user_message)

        # 2. ж №жҚ®ж„Ҹеӣҫи·Ҝз”ұеҲ°зӣёеә”еӨ„зҗҶеҷЁ
        if intent == "code_editing":
            return await self.handle_code_editing(user_message)
        elif intent == "file_operations":
            return await self.handle_file_operations(user_message)
        elif intent == "web_search":
            return await self.handle_web_search(user_message)
        elif intent == "terminal":
            return await self.handle_terminal(user_message)
        else:
            return await self.handle_unknown(user_message)

        # 3. дҪҝз”Ё LLM з”ҹжҲҗзј–иҫ‘

        return AgentResponse(
            message=f"е·ІеҜ№ {file_path} иҝӣиЎҢ {len(edits)} еӨ„зј–иҫ‘",
            edits=edits
        )

class KnowledgeShare:
    """зҹҘиҜҶе…ұдә«з»„д»¶"""

    def __init__(self, faiss: FAISSIndex):
        self.faiss = faiss
        self.documents = {}

    def share_knowledge(self, document: Document) -> ShareResult:
        """еҲҶдә«зҹҘиҜҶ"""
        # 1. ж·»еҠ еҲ°зҙўеј•
        self.documents[document.id] = document

        # 2. жӣҙж–° FAISS зҙўеј•
        self.faiss.add(document.embedding)

        # 3. дҪҝе…¶еҸҜиў«жҗңзҙў
        return ShareResult(
            document_id=document.id,
            searchable=True
        )

    async def query_knowledge(self, query: str, k: int = 5) -> QueryResult:
        """жҹҘиҜўзҹҘиҜҶ"""
        # 1. иҪ¬жҚўжҹҘиҜўдёәеҗ‘йҮҸ
        query_vector = await self._embed(query)

        # 2. жҗңзҙўжңҖиҝ‘йӮ»
        indices, distances = self.faiss.search(query_vector, k)

        # 3. иҺ·еҸ–ж–ҮжЎЈ
        results = [self.documents[i] for i in indices]

        return QueryResult(results=results, distances=distances)
